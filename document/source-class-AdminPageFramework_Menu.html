367     /**
368      * Builds the sidebar menu of the added pages.
369      * 
370      * @since           2.0.0
371      * @internal
372      */
373     public function _replyToBuildMenu() {
374         
375         // If the root menu label is not set but the slug is set, 
376         if ( $this->oProp->aRootMenu['fCreateRoot'] ) 
377             $this->_registerRootMenuPage();
378         
379         // Apply filters to let other scripts add sub menu pages.
380         $this->oProp->aPages = $this->oUtil->addAndApplyFilter(     // Parameters: $oCallerObject, $sFilter, $vInput, $vArgs...
381             $this,
382             "pages_{$this->oProp->sClassName}", 
383             $this->oProp->aPages
384         );
385         
386         // Sort the page array.
387         uasort( $this->oProp->aPages, array( $this, '_sortByOrder' ) ); 
388         
389         // Set the default page, the first element.
390         foreach ( $this->oProp->aPages as $aPage ) {
391             
392             if ( ! isset( $aPage['page_slug'] ) ) continue;
393             $this->oProp->sDefaultPageSlug = $aPage['page_slug'];
394             break;
395             
396         }
397         
398         // Register them.
399         foreach ( $this->oProp->aPages as &$aSubMenuItem ) {
400             $aSubMenuItem = $this->_formatSubMenuItemArray( $aSubMenuItem );    // needs to be sanitized because there are hook filters applied to this array.
401             $aSubMenuItem['_page_hook'] = $this->_registerSubMenuItem( $aSubMenuItem ); // store the page hook; this is same as the value stored in the global $page_hook or $hook_suffix variable. 
402         }
403                         
404         // After adding the sub menus, if the root menu is created, remove the page that is automatically created when registering the root menu.
405         if ( $this->oProp->aRootMenu['fCreateRoot'] ) 
406             remove_submenu_page( $this->oProp->aRootMenu['sPageSlug'], $this->oProp->aRootMenu['sPageSlug'] );
407         
408     }

419                 $this->oProp->aRootMenu['sTitle'],              // Menu title - should be the root page title.
420                 $this->oProp->sCapability,                      // Capability - access right
421                 $this->oProp->aRootMenu['sPageSlug'],           // Menu ID 
422                 '', //array( $this, $this->oProp->sClassName ),     // Page content displaying function - the root page will be removed so no need to register a function.
423                 $this->oProp->aRootMenu['sIcon16x16'],      // icon path
424                 isset( $this->oProp->aRootMenu['iPosition'] ) ? $this->oProp->aRootMenu['iPosition'] : null // menu position
425             );

445         /**
446          * Formats the given sub-menu link array.
447          * @since           3.0.0
448          * @internal
449          */
450         private function _formatSubmenuLinkArray( $aSubMenuLink ) {
451             
452             // If the set URL is not valid, return.
453             if ( ! filter_var( $aSubMenuLink['href'], FILTER_VALIDATE_URL ) ) return array();
454             
455             return $this->oUtil->uniteArrays(       
456                 array(  
457                     'capability'    => isset( $aSubMenuLink['capability'] ) ? $aSubMenuLink['capability'] : $this->oProp->sCapability,
458                     'order'         => isset( $aSubMenuLink['order'] ) && is_numeric( $aSubMenuLink['order'] ) ? $aSubMenuLink['order'] : count( $this->oProp->aPages ) + 10,
459                 ),
460                 $aSubMenuLink + self::$_aStructure_SubMenuLinkForUser
461             );          
462             
463         }
464         /**
465          * Formats the given sub-menu page array.
466          * @since           3.0.0
467          * @internal
468          */
469         private function _formatSubMenuPageArray( $aSubMenuPage ) {
470             
471             $aSubMenuPage = $aSubMenuPage + self::$_aStructure_SubMenuPageForUser;
472 
473             $aSubMenuPage['screen_icon_id'] = trim( $aSubMenuPage['screen_icon_id'] );          
474             return $this->oUtil->uniteArrays(
475                 array(

493         /**
494          * Registers the sub-menu item.
495          * 
496          * @since           2.0.0
497          * @since           3.0.0           Changed the name from registerSubMenuPage().
498          * @remark          Used in the buildMenu() method. 
499          * @remark          Within the <em>admin_menu</em> hook callback process.
500          * @remark          The sub menu page slug should be unique because add_submenu_page() can add one callback per page slug.
501          * @internal
502          */ 
503         private function _registerSubMenuItem( $aArgs ) {
504                 
505             // Variables
506             $sType = $aArgs['type'];    // page or link
507             $sTitle = $sType == 'page' ? $aArgs['title'] : $aArgs['title'];
508             $sCapability = isset( $aArgs['capability'] ) ? $aArgs['capability'] : $this->oProp->sCapability;
509             $_sPageHook = '';
510             
511             // Check the capability
512             if ( ! current_user_can( $sCapability ) ) return;       
513             
514             // Add the sub-page to the sub-menu
515             
516             $sRootPageSlug = $this->oProp->aRootMenu['sPageSlug'];
517             $sMenuLabel = plugin_basename( $sRootPageSlug );    // Make it compatible with the add_submenu_page() function.
518             
519             // If it's a page - it's possible that the page_slug key is not set if the user uses a method like setPageHeadingTabsVisibility() prior to addSubMenuItam().
520             if ( $sType == 'page' && isset( $aArgs['page_slug'] ) ) {       
521                 
522                 $sPageSlug = $aArgs['page_slug'];
523                 $_sPageHook = add_submenu_page( 
524                     $sRootPageSlug,                     // the root(parent) page slug
525                     $sTitle,                                // page_title
526                     $sTitle,                                // menu_title
527                     $sCapability,                           // capability
528                     $sPageSlug, // menu_slug
529                     // In admin.php ( line 149 of WordPress v3.6.1 ), do_action($page_hook) ( where $page_hook is $_sPageHook )
530                     // will be executed and it triggers the __call magic method with the method name of "md5 class hash + _page_ + this page slug".
531                     array( $this, $this->oProp->sClassHash . '_page_' . $sPageSlug )
532                 );          
533                 
534                 add_action( "load-" . $_sPageHook , array( $this, "load_pre_" . $sPageSlug ) );
535 
536                 // If the visibility option is false, remove the one just added from the sub-menu array
537                 if ( ! $aArgs['show_in_menu'] ) {
538 
539                     foreach( ( array ) $GLOBALS['submenu'][ $sMenuLabel ] as $iIndex => $aSubMenu ) {
540                         
541                         if ( ! isset( $aSubMenu[ 3 ] ) ) continue;
542                         
543                         // the array structure is defined in plugin.php - $submenu[$parent_slug][] = array ( $menu_title, $capability, $menu_slug, $page_title ) 
544                         if ( $aSubMenu[0] == $sTitle && $aSubMenu[3] == $sTitle && $aSubMenu[2] == $sPageSlug ) {
545                             unset( $GLOBALS['submenu'][ $sMenuLabel ][ $iIndex ] );
546                             
547                             // The page title in the browser window title bar will miss the page title as this is left as it is.
548                             $this->oProp->aHiddenPages[ $sPageSlug ] = $sTitle;
549                             add_filter( 'admin_title', array( $this, '_replyToFixPageTitleForHiddenPages' ), 10, 2 );
550                             
551                             break;
552                         }
553                     }
554                 } 
555                     
556             } 
557             // If it's a link,
558             if ( $sType == 'link' && $aArgs['show_in_menu'] ) {
559                 
560                 if ( ! isset( $GLOBALS['submenu'][ $sMenuLabel ] ) )
561                     $GLOBALS['submenu'][ $sMenuLabel ] = array();
562                 
563                 $GLOBALS['submenu'][ $sMenuLabel ][] = array ( 
564                     $sTitle, 
565                     $sCapability, 
566                     $aArgs['href'],
567                 );  
568             }
569         
570             return $_sPageHook; // will be stored in the $this->oProp->aPages property.
571 
572         }
