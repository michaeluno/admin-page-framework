170     /**
171     * Adds a custom taxonomy to the class post type.
172     * <h4>Example</h4>
173     * <code>$this->addTaxonomy( 
174     * 'sample_taxonomy', // taxonomy slug
175     * array( // argument
176     * 'labels' => array(
177     * 'name' => 'Genre',
178     * 'add_new_item' => 'Add New Genre',
179     * 'new_item_name' => "New Genre"
180     * ),
181     * 'show_ui' => true,
182     * 'show_tagcloud' => false,
183     * 'hierarchical' => true,
184     * 'show_admin_column' => true,
185     * 'show_in_nav_menus' => true,
186     * 'show_table_filter' => true, // framework specific key
187     * 'show_in_sidebar_menus' => false, // framework specific key
188     * )
189     * );</code>
190     * 
191     * @see http://codex.wordpress.org/Function_Reference/register_taxonomy#Arguments
192     * @since 2.0.0
193     * @since 3.1.1 Added the third parameter.
194     * @param string $sTaxonomySlug The taxonomy slug.
195     * @param array $aArgs The taxonomy argument array passed to the second parameter of the <a href="http://codex.wordpress.org/Function_Reference/register_taxonomy#Arguments">register_taxonomy()</a> function.
196     * @param array $aAdditionalObjectTypes Additional object types(post types) besides the caller post type.
197     * @return void
198     */ 
199     protected function addTaxonomy( $sTaxonomySlug, array $aArgs, array $aAdditionalObjectTypes=array() ) {
200         
201         $sTaxonomySlug = $this->oUtil->sanitizeSlug( $sTaxonomySlug );
202         $this->oProp->aTaxonomies[ $sTaxonomySlug ] = $aArgs;    
203         if ( isset( $aArgs['show_table_filter'] ) && $aArgs['show_table_filter'] ) {
204             $this->oProp->aTaxonomyTableFilters[] = $sTaxonomySlug;
205         }
206         if ( isset( $aArgs['show_in_sidebar_menus'] ) && ! $aArgs['show_in_sidebar_menus'] ) {
207             $this->oProp->aTaxonomyRemoveSubmenuPages[ "edit-tags.php?taxonomy={$sTaxonomySlug}&amp;post_type={$this->oProp->sPostType}" ] = "edit.php?post_type={$this->oProp->sPostType}";
208         }
209         
210         $_aExistingObjectTypes = isset( $this->oProp->aTaxonomyObjectTypes[ $sTaxonomySlug ] ) && is_array( $this->oProp->aTaxonomyObjectTypes[ $sTaxonomySlug ] )
211             ? $this->oProp->aTaxonomyObjectTypes[ $sTaxonomySlug ] 
212             : array();
213         $aAdditionalObjectTypes = array_merge( $_aExistingObjectTypes, $aAdditionalObjectTypes );
214         $this->oProp->aTaxonomyObjectTypes[ $sTaxonomySlug ] = array_unique( $aAdditionalObjectTypes );
215 
216         // Set up hooks.
217         if ( count( $this->oProp->aTaxonomyTableFilters ) == 1 ) {
218             if ( did_action( 'init' ) ) {
219                 $this->_replyToRegisterTaxonomies();
220             } else {
221                 add_action( 'init', array( $this, '_replyToRegisterTaxonomies' ) ); // the hook should not be admin_init because taxonomies need to be accessed in regular pages.
222             }
223         }
224         if ( count( $this->oProp->aTaxonomyRemoveSubmenuPages ) == 1 ) {
225             if ( did_action( 'admin_menu' ) ) {
226                 $this->_replyToRemoveTexonomySubmenuPages();
227             } else {
228                 add_action( 'admin_menu', array( $this, '_replyToRemoveTexonomySubmenuPages' ), 999 ); 
229             }
230         }
