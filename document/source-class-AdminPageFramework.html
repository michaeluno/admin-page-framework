 11  * <p>The class automatically creates WordPress action and filter hooks associated with the class methods.

107  * <h3>Remarks</h3>
108  * <p>The slugs must not contain a dot(.) or a hyphen(-) since it is used in the callback method name.</p>
109  * <h3>Examples</h3>
110  * <p>If the instantiated class name is Sample_Admin_Pages, defining the following class method will embed a banner image in all pages created by the class.</p>
111  * <code>class Sample_Admin_Pages extends AdminPageFramework {
112  * ...
113  *     function content_top_Sample_Admin_Pages( $sContent ) {
114  *         return '<div style="float:right;"><img src="' . plugins_url( 'img/banner468x60.gif', __FILE__ ) . '" /></div>' 
115  *             . $sContent;
116  *     }
117  * ...
118  * }</code>
119  * <p>If the created page slug is my_first_setting_page, defining the following class method will filter the middle part of the page output.</p>
120  * <code>class Sample_Admin_Pages extends AdminPageFramework {
121  * ...
122  *     function content_my_first_setting_page( $sContent ) {
123  *         return $sContent . '<p>Hello world!</p>';

614     /**
615      * The magic method which redirects callback-function calls with the pre-defined prefixes for hooks to the appropriate methods. 
616      * 
617      * @access          public
618      * @remark          the users do not need to call or extend this method unless they know what they are doing.
619      * @param           string      the called method name. 
620      * @param           array       the argument array. The first element holds the parameters passed to the called method.
621      * @return          mixed       depends on the called method. If the method name matches one of the hook prefixes, the redirected methods return value will be returned. Otherwise, none.
622      * @since           2.0.0
623      * @internal
624      */
625     public function __call( $sMethodName, $aArgs=null ) {       
626                  
627         // The currently loading in-page tab slug. Be careful that not all cases $sMethodName have the page slug.
628         $sPageSlug = isset( $_GET['page'] ) ? $_GET['page'] : null; 
629         $sTabSlug = isset( $_GET['tab'] ) ? $_GET['tab'] : $this->oProp->getDefaultInPageTab( $sPageSlug ); 
630 
631         // If it is a pre callback method, call the redirecting method.     
632         if ( substr( $sMethodName, 0, strlen( 'section_pre_' ) )    == 'section_pre_' ) return $this->_renderSectionDescription( $sMethodName );  // add_settings_section() callback    - defined in AdminPageFramework_Setting
633         if ( substr( $sMethodName, 0, strlen( 'field_pre_' ) )      == 'field_pre_' )   return $this->_renderSettingField( $aArgs[ 0 ], $sPageSlug );  // add_settings_field() callback - defined in AdminPageFramework_Setting
634         if ( substr( $sMethodName, 0, strlen( 'validation_pre_' ) ) == 'validation_pre_' )  return $this->_doValidationCall( $sMethodName, $aArgs[ 0 ] ); // register_setting() callback - defined in AdminPageFramework_Setting
635         if ( substr( $sMethodName, 0, strlen( 'load_pre_' ) )       == 'load_pre_' )    return $this->_doPageLoadCall( substr( $sMethodName, strlen( 'load_pre_' ) ), $sTabSlug, $aArgs[ 0 ] );  // load-{page} callback
636 
637         // The callback of the call_page_{page slug} action hook
638         if ( $sMethodName == $this->oProp->sClassHash . '_page_' . $sPageSlug )
639             return $this->_renderPage( $sPageSlug, $sTabSlug );     // the method is defined in the AdminPageFramework_Page class.
640         
641         // If it's one of the framework's callback methods, do nothing. 
642         if ( $this->_isFrameworkCallbackMethod( $sMethodName ) )
643             return isset( $aArgs[0] ) ? $aArgs[0] : null;   // if $aArgs[0] is set, it's a filter, otherwise, it's an action.       
644         
645     }   
646         /**
647          * Determines whether the method name matches the pre-defined hook prefixes.
648          * @access          private
649          * @since           2.0.0
650          * @remark          the users do not need to call or extend this method unless they know what they are doing.
651          * @param           string          $sMethodName            the called method name
652          * @return          boolean         If it is a framework's callback method, returns true; otherwise, false.
653          * @internal
654          */
655         private function _isFrameworkCallbackMethod( $sMethodName ) {
656                 
657             foreach( self::$_aHookPrefixes as $sPrefix ) 
658                 if ( substr( $sMethodName, 0, strlen( $sPrefix ) )  == $sPrefix  ) 
659                     return true;
660             
661             return false;
662             
663         }
