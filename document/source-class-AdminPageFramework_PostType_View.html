 61         
 62         $_sLinkLabel = isset( $this->oProp->aPostTypeArgs['labels']['plugin_listing_table_title_cell_link'] )
 63             ? $this->oProp->aPostTypeArgs['labels']['plugin_listing_table_title_cell_link']
 64             : $this->oMsg->__( 'manage' );
 65             
 66         // If the user explicitly sets an empty string to the label key, do not insert a link.
 67         if ( ! $_sLinkLabel ) {
 68             return $aLinks;
 69         }
 70                         
 71         // http://.../wp-admin/edit.php?post_type=[...]
 72         array_unshift(  
 73             $aLinks,
 74             "<a href='" . esc_url( "edit.php?post_type={$this->oProp->sPostType}" ) . "'>" . $_sLinkLabel . "</a>"
 75         );

107     /**
108      * Adds drop-down lists to filter posts by added taxonomies, placed above the post type listing table.
109      * 
110      * @internal
111      */ 
112     public function _replyToAddTaxonomyTableFilter() {
113         
114         if ( $GLOBALS['typenow'] != $this->oProp->sPostType ) return;
115         
116         // If there is no post added to the post type, do nothing.
117         $oPostCount = wp_count_posts( $this->oProp->sPostType );
118         if ( $oPostCount->publish + $oPostCount->future + $oPostCount->draft + $oPostCount->pending + $oPostCount->private + $oPostCount->trash == 0 )
119             return;
120         
121         foreach ( get_object_taxonomies( $GLOBALS['typenow'] ) as $sTaxonomySulg ) {
122             
123             if ( ! in_array( $sTaxonomySulg, $this->oProp->aTaxonomyTableFilters ) ) continue;
124             
125             $oTaxonomy = get_taxonomy( $sTaxonomySulg );
126  
127             // If there is no added term, skip.
128             if ( wp_count_terms( $oTaxonomy->name ) == 0 ) continue;            
129 
130             // This function will echo the drop down list based on the passed array argument.
131             wp_dropdown_categories( array(
132                 'show_option_all' => $this->oMsg->__( 'show_all' ) . ' ' . $oTaxonomy->label,
133                 'taxonomy'    => $sTaxonomySulg,
134                 'name'        => $oTaxonomy->name,
135                 'orderby'     => 'name',
136                 'selected'    => intval( isset( $_GET[ $sTaxonomySulg ] ) ),
137                 'hierarchical'    => $oTaxonomy->hierarchical,
138                 'show_count'      => true,
139                 'hide_empty'      => false,
140                 'hide_if_empty' => false,
141                 'echo'  => true,    // this make the function print the output
142             ) );
143             
144         }
145     }
146     /**
147      * Returns a query object based on the taxonomies belongs to the post type.
148      * 
149      * @internal
150      */
151     public function _replyToGetTableFilterQueryForTaxonomies( $oQuery=null ) {
152         
153         if ( 'edit.php' != $this->oProp->sPageNow ) return $oQuery;
154         
155         if ( ! isset( $GLOBALS['typenow'] ) ) return $oQuery;
156 
157         foreach ( get_object_taxonomies( $GLOBALS['typenow'] ) as $sTaxonomySlug ) {
158             
159             if ( ! in_array( $sTaxonomySlug, $this->oProp->aTaxonomyTableFilters ) ) continue;
160             
161             $sVar = &$oQuery->query_vars[ $sTaxonomySlug ];
162             if ( ! isset( $sVar ) ) continue;
163             
164             $oTerm = get_term_by( 'id', $sVar, $sTaxonomySlug );
165             if ( is_object( $oTerm ) )
166                 $sVar = $oTerm->slug;
167 
168         }
169         
170         return $oQuery;
171         
172     }

179     public function _replyToPrintStyle() {
180         
181         if ( $this->oUtil->getCurrentPostType() !== $this->oProp->sPostType ) {
182             return;
183         }
184 
185         // If the screen icon url is specified
186         if ( isset( $this->oProp->aPostTypeArgs['screen_icon'] ) && $this->oProp->aPostTypeArgs['screen_icon'] ) {
187             $this->oProp->sStyle .= $this->_getStylesForPostTypeScreenIcon( $this->oProp->aPostTypeArgs['screen_icon'] );
188         }
189             
190         $this->oProp->sStyle = $this->oUtil->addAndApplyFilters( $this, "style_{$this->oProp->sClassName}", $this->oProp->sStyle );
191         
192         // Print out the filtered styles.
193         if ( ! empty( $this->oProp->sStyle ) ) {
194             echo "<style type='text/css' id='admin-page-framework-style-post-type'>" 
195                 . $this->oProp->sStyle
196                 . "</style>";           
197         }
