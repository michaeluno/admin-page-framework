 69     /**
 70      * Adds drop-down lists to filter posts by added taxonomies, placed above the post type listing table.
 71      * 
 72      * @internal
 73      */ 
 74     public function _replyToAddTaxonomyTableFilter() {
 75         
 76         if ( $GLOBALS['typenow'] != $this->oProp->sPostType ) { return; }
 77         
 78         // If there is no post added to the post type, do nothing.
 79         $oPostCount = wp_count_posts( $this->oProp->sPostType );
 80         if ( $oPostCount->publish + $oPostCount->future + $oPostCount->draft + $oPostCount->pending + $oPostCount->private + $oPostCount->trash == 0 ) {
 81             return;
 82         }
 83         
 84         foreach ( get_object_taxonomies( $GLOBALS['typenow'] ) as $sTaxonomySulg ) {
 85             
 86             if ( ! in_array( $sTaxonomySulg, $this->oProp->aTaxonomyTableFilters ) ) continue;
 87             
 88             $oTaxonomy = get_taxonomy( $sTaxonomySulg );
 89  
 90             // If there is no added term, skip.
 91             if ( wp_count_terms( $oTaxonomy->name ) == 0 ) continue;             
 92 
 93             // This function will echo the drop down list based on the passed array argument.
 94             wp_dropdown_categories( array(
 95                 'show_option_all' => $this->oMsg->__( 'show_all' ) . ' ' . $oTaxonomy->label,
 96                 'taxonomy'       => $sTaxonomySulg,
 97                 'name'           => $oTaxonomy->name,
 98                 'orderby'       => 'name',
 99                 'selected'       => intval( isset( $_GET[ $sTaxonomySulg ] ) ),
100                 'hierarchical'       => $oTaxonomy->hierarchical,
101                 'show_count'       => true,
102                 'hide_empty'       => false,
103                 'hide_if_empty' => false,
104                 'echo' => true, // this make the function print the output
105             ) );
106             
107         }
108     }
109     /**
110      * Returns a query object based on the taxonomies belongs to the post type.
111      * 
112      * @internal
113      */
114     public function _replyToGetTableFilterQueryForTaxonomies( $oQuery=null ) {
115         
116         if ( 'edit.php' != $this->oProp->sPageNow ) { return $oQuery; }
117         
118         if ( ! isset( $GLOBALS['typenow'] ) ) { return $oQuery; }
119 
120         foreach ( get_object_taxonomies( $GLOBALS['typenow'] ) as $sTaxonomySlug ) {
121             
122             if ( ! in_array( $sTaxonomySlug, $this->oProp->aTaxonomyTableFilters ) ) { continue; }
123             
124             $sVar = &$oQuery->query_vars[ $sTaxonomySlug ];
125             if ( ! isset( $sVar ) ) { continue; }
126             
127             $oTerm = get_term_by( 'id', $sVar, $sTaxonomySlug );
128             if ( is_object( $oTerm ) ) {
129                 $sVar = $oTerm->slug;
130             }
131 
132         }
133         
134         return $oQuery;
135         
136     }

144

147         }
148 
149         // If the screen icon url is specified
150         if ( isset( $this->oProp->aPostTypeArgs['screen_icon'] ) && $this->oProp->aPostTypeArgs['screen_icon'] ) {
151             $this->oProp->sStyle .= $this->_getStylesForPostTypeScreenIcon( $this->oProp->aPostTypeArgs['screen_icon'] );
152         }
153             
154         $this->oProp->sStyle = $this->oUtil->addAndApplyFilters( $this, "style_{$this->oProp->sClassName}", $this->oProp->sStyle );
155         
156         // Print out the filtered styles.
157         if ( ! empty( $this->oProp->sStyle ) ) {
158             echo "<style type='text/css' id='admin-page-framework-style-post-type'>" 
159                 . $this->oProp->sStyle
160                 . "</style>";     
161         }

201      * Filters the post type post content.
202      *

205     public function _replyToFilterPostTypeContent( $sContent ) {
206         
207         if ( ! is_singular() ) {
208             return $sContent;
209         }
210         if ( ! is_main_query() ) {
211             return $sContent;
212         }
213         global $post;
214         if ( $this->oProp->sPostType !== $post->post_type ) {
215             return $sContent;
216         }
217     
218         return $this->oUtil->addAndApplyFilters(
219             $this, 
220             "content_{$this->oProp->sClassName}", 
221             $this->content( $sContent )
222         );
