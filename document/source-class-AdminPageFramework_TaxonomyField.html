159     /**
160      * Retrieves the fields output.
161      * 
162      * @since           3.0.0
163      * @internal
164      */
165     private function _getFieldsOutput( $iTermID, $bRenderTableRow ) {
166     
167         $aOutput = array();
168         
169         /* Set nonce. */
170         $aOutput[] = wp_nonce_field( $this->oProp->sClassHash, $this->oProp->sClassHash, true, false );
171         
172         /* Set the option property array */
173         $this->setOptionArray( $iTermID, $this->oProp->sOptionKey );
174 
175         foreach ( $this->oProp->aFields as $aField ) {
176             
177             // Avoid undefined index warnings
178             $aField = array( '_field_type' => 'taxonomy' ) + $aField + AdminPageFramework_Property_MetaBox::$_aStructure_Field;
179             
180             // Check capability. If the access level is not sufficient, skip.
181             $aField['capability'] = isset( $aField['capability'] ) ? $aField['capability'] : $this->oProp->sCapability;
182             if ( ! current_user_can( $aField['capability'] ) ) continue;            
183 
184             // If a custom condition is set and it's not true, skip.
185             if ( ! $aField['if'] ) continue;
186         
187             if ( $bRenderTableRow ) :
188                 $aOutput[] = "<tr>";    // Begin a table row. 
189                     if ( $aField['show_title_column'] )
190                         $aOutput[] = 
191                             "<th>"
192                                 ."<label for='{$aField['field_id']}'>"
193                                     . "<a id='{$aField['field_id']}'></a>"
194                                         . "<span title='" . strip_tags( isset( $aField['tip'] ) ? $aField['tip'] : $aField['description'] ) . "'>"
195                                             . $aField['title'] 
196                                         . "</span>"
197                                 . "</label>"
198                             . "</th>";      
199                     $aOutput[] = "<td>";
200                         $aOutput[] = $this->getFieldOutput( $aField );
201                     $aOutput[] = "</td>";
202                 $aOutput[] = "</tr>";
203             else :              
204                 if ( $aField['show_title_column'] )
205                     $aOutput[] = 
206                         "<label for='{$aField['field_id']}'>"
207                             . "<a id='{$aField['field_id']}'></a>"
208                                 . "<span title='" . strip_tags( isset( $aField['tip'] ) ? $aField['tip'] : $aField['description'] ) . "'>"
209                                     . $aField['title'] 
210                                 . "</span>"
211                         . "</label>";                   
212                 $aOutput[] = $this->getFieldOutput( $aField );
213             endif;
214 
215             
216         } // end foreach
217         
218         /* Filter the output */
219         $sOutput = $this->oUtil->addAndApplyFilters( $this, 'content_' . $this->oProp->sClassName, implode( PHP_EOL, $aOutput ) );
220         
221         /* Do action */
222         $this->oUtil->addAndDoActions( $this, 'do_' . $this->oProp->sClassName );
223             
224         return $sOutput;    
225     
226     }

234     public function _replyToValidateOptions( $iTermID ) {
235         
236         if ( ! wp_verify_nonce( $_POST[ $this->oProp->sClassHash ], $this->oProp->sClassHash ) ) return;
237         
238         $aTaxonomyFieldOptions = get_option( $this->oProp->sOptionKey );
239         $aOldOptions = isset( $aTaxonomyFieldOptions[ $iTermID ] ) ? $aTaxonomyFieldOptions[ $iTermID ] : array();
240         $aSubmittedOptions = array();
241         foreach( array_keys( $this->oProp->aFields ) as $sFieldID ) 
242             if ( isset( $_POST[ $sFieldID ] ) ) $aSubmittedOptions[ $sFieldID ] = $_POST[ $sFieldID ];
243 
244         /* Apply validation filters to the submitted option array. */
245         $aSubmittedOptions = $this->oUtil->addAndApplyFilters( $this, 'validation_' . $this->oProp->sClassName, $aSubmittedOptions, $aOldOptions );
246         
247         $aTaxonomyFieldOptions[ $iTermID ] = $this->oUtil->uniteArrays( $aSubmittedOptions, $aOldOptions );
248         update_option( $this->oProp->sOptionKey, $aTaxonomyFieldOptions );
249         
250     }
