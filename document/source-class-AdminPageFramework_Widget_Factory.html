 11  * A back-end factory class that redirects callback methods to the main widget class.

 18 class AdminPageFramework_Widget_Factory extends WP_Widget {
 19     
 20     /**
 21      * Sets up internal properties.
 22      */
 23     public function __construct( $oCaller, $sWidgetTitle, array $aArguments=array() ) {
 24         
 25         $aArguments = $aArguments 
 26             + array( 
 27                 'classname'     => 'admin_page_framework_widget',
 28                 'description'   => '',  
 29             );
 30         parent::__construct( 
 31             $oCaller->oProp->sClassName,  // base id 
 32             $sWidgetTitle,      // widget title
 33             $aArguments         // widget arguments
 34         );
 35         $this->oCaller = $oCaller;
 36         
 37     }

 39     /**
 40      * Displays the widget contents in the front end.
 41      */
 42     public function widget( $aArguments, $aFormData ) {
 43 
 44         echo $aArguments['before_widget'];
 45         
 46         $_sTitle = apply_filters( 'widget_title', isset( $aFormData['title'] ) ? $aFormData['title'] : '', $aFormData, $this->id_base );
 47         if ( $_sTitle ) {
 48             echo $aArguments['before_title'] . $_sTitle . $aArguments['after_title'];
 49         }
 50 
 51         // Do action.
 52         $this->oCaller->oUtil->addAndDoActions( $this->oCaller, 'do_' . $this->oCaller->oProp->sClassName, $this->oCaller );
 53         
 54         // Filter the contents.
 55         echo $this->oCaller->oUtil->addAndApplyFilters(
 56             $this->oCaller, 
 57             "content_{$this->oCaller->oProp->sClassName}", 
 58             $this->oCaller->content( '', $aArguments, $aFormData ),
 59             $aArguments,
 60             $aFormData
 61         );    
 62  
 63         echo $aArguments['after_widget'];
 64         
 65     }
 66     
 67     /**
 68      * Validates the submitted form data.
 69      */
 70     public function update( $aSubmittedFormData, $aSavedFormData ) {
 71                 
 72         return $this->oCaller->oUtil->addAndApplyFilters(
 73             $this->oCaller, 
 74             "validation_{$this->oCaller->oProp->sClassName}", 
 75             $this->oCaller->validate( $aSubmittedFormData, $aSavedFormData, $this->oCaller ),
 76             $aSavedFormData,
 77             $this->oCaller
 78         );
 79         
 80     }

 82     /**
 83      * Constructs the widget form.
 84      */
 85     public function form( $aFormData ) {
 86 
 87         // Trigger the load() method and load_{...} actions. The user sets up the form.
 88         $this->oCaller->load( $this->oCaller );
 89         $this->oCaller->oUtil->addAndDoActions( 
 90             $this->oCaller, 
 91             'load_' . $this->oCaller->oProp->sClassName, 
 92             $this->oCaller 
 93         );
 94             
 95         // Register the form elements.
 96         $this->oCaller->_registerFormElements( $aFormData );
 97       
 98         // Set up callbacks for field element outputs such as for name and it attributes.
 99         $this->oCaller->oProp->aFieldCallbacks = array( 
100             'hfID'          => array( $this, 'get_field_id' ),    // defined in the WP_Widget class.  
101             'hfTagID'       => array( $this, 'get_field_id' ),    // defined in the WP_Widget class.  
102             'hfName'        => array( $this, 'get_field_name' ),  // defined in the WP_Widget class.
103             // 'hfClass'       => array( $this, '_replyToAddClassSelector' ),
104             // 'hfNameFlat'    => array( $this, '' ),
105         );              
106       
107         // Render the form. 
108         $this->oCaller->_printWidgetForm();
109        
110     }
