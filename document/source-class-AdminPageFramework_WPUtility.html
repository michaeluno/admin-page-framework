 13     /**
 14      * Triggers the do_action() function with the given action names and the arguments.
 15      * 
 16      * This is useful to perform do_action() on multiple action hooks with the same set of arguments.
 17      * For example, if there are the following action hooks, <em>action_name</em>, <em>action_name1</em>, and <em>action_name2</em>, and to perform these, normally it takes the following lines.
 18      * <code>do_action( 'action_name1', $var1, $var2 );
 19      * do_action( 'action_name2', $var1, $var2 );
 20      * do_action( 'action_name3', $var1, $var2 );</code>
 21      * 
 22      * This method saves these line this way:
 23      * <code>$this->doActions( array( 'action_name1', 'action_name2', 'action_name3' ), $var1, $var2 );</code>
 24      * 
 25      * <h4>Example</h4>
 26      * <code>$this->doActions( array( 'action_name1' ), $var1, $var2, $var3 );</code> 
 27      * 
 28      * @since           2.0.0
 29      * @access          public
 30      * @remark          Accepts variadic parameters; the number of accepted parameters are not limited to four.
 31      * @param           array           $aActionHooks           a numerically indexed array consisting of action hook names to execute.
 32      * @param           mixed           $vArgs1                 an argument to pass to the action callbacks.
 33      * @param           mixed           $vArgs2                 another argument to pass to the action callbacks.
 34      * @param           mixed           $_and_more              add as many arguments as necessary to the next parameters.
 35      * @return          void            does not return a value.
 36      */     
 37     public function doActions( $aActionHooks, $vArgs1=null, $vArgs2=null, $_and_more=null ) {
 38         
 39         $aArgs = func_get_args();       
 40         $aActionHooks = $aArgs[ 0 ];
 41         foreach( ( array ) $aActionHooks as $sActionHook  ) {
 42             $aArgs[ 0 ] = $sActionHook;
 43             call_user_func_array( 'do_action' , $aArgs );
 44         }

 49      * Adds the method of the given action hook name(s) to the given action hook(s) with arguments.
 50      * 
 51      * In other words, this enables to register methods to the custom hooks with the same name and triggers the callbacks (not limited to the registered ones) assigned to the hooks. 
 52      * Of course, the registered methods will be triggered right away. Thus, the magic overloading __call() should catch them and redirect the call to the appropriate methods.
 53      * This enables, at the same time, publicly the added custom action hooks; therefore, third-party scripts can use the action hooks.
 54      * 
 55      * This is the reason the object instance must be passed to the first parameter. Regular functions as the callback are not supported for this method.
 56      * 
 57      * <h4>Example</h4>
 58      * <code>$this->oUtil->addAndDoActions( $this, array( 'my_action1', 'my_action2', 'my_action3' ), 'argument_a', 'argument_b' );</code>
 59      * 
 60      * @access          public
 61      * @since           2.0.0
 62      * @remark          Accepts variadic parameters.
 63      * @param           object          $oCallerObject          the object that holds the callback method that matches the action hook name.
 64      * @param           array           $aActionHooks           a numerically index array consisting of action hook names that serve as the callback method names. 
 65      * @param           mixed           $vArgs1                 the argument to pass to the hook callback functions.
 66      * @param           mixed           $vArgs2                 another argument to pass to the hook callback functions.
 67      * @param           mixed           $_and_more              add as many arguments as necessary to the next parameters.
 68      * @return          void
 69      */ 
 70     public function addAndDoActions( $oCallerObject, $aActionHooks, $vArgs1=null, $vArgs2=null, $_and_more=null ) {
 71

 76             $aArgs[ 1 ] = $sActionHook;

273      * @since           2.1.5
274      * @static
275      * @access          public
276      * @return          string          The source url
277      */
278     static public function getSRCFromPath( $sFilePath ) {
279                         
280         $oWPStyles = new WP_Styles();   // It doesn't matter whether the file is a style or not. Just use the built-in WordPress class to calculate the SRC URL.
281         $sRelativePath = AdminPageFramework_Utility::getRelativePath( ABSPATH, $sFilePath );        
282         $sRelativePath = preg_replace( "/^\.[\/\\\]/", '', $sRelativePath, 1 ); // removes the heading ./ or .\ 
283         $sHref = trailingslashit( $oWPStyles->base_url ) . $sRelativePath;
284         unset( $oWPStyles );    // for PHP 5.2.x or below
285         return esc_url( $sHref );

289     /**
290      * Resolves the given src.
291      * 
292      * Checks if the given string is a url, a relative path, or an absolute path and returns the url if it's not a relative path.
293      * 
294      * @since           2.1.5
295      * @since           2.1.6           Moved from the AdminPageFramework_HeadTag_Base class. Added the $bReturnNullIfNotExist parameter.
296      */
297     static public function resolveSRC( $sSRC, $bReturnNullIfNotExist=false ) {  
298 
299         if ( ! $sSRC )  
300             return $bReturnNullIfNotExist ? null : $sSRC;
301             
302         // It is a url
303         if ( filter_var( $sSRC, FILTER_VALIDATE_URL ) )
304             return $sSRC;
305 
306         // If the file exists, it means it is an absolute path. If so, calculate the URL from the path.
307         if ( file_exists( realpath( $sSRC ) ) )
308             return self::getSRCFromPath( $sSRC );
309         
310         if ( $bReturnNullIfNotExist )
311             return null;
312         
313         // Otherwise, let's assume the string is a relative path 'to the WordPress installed absolute path'.
314         return $sSRC;
315         
316     }
