 58     /**
 59      * Determines whether the currently loaded page is of the post type page.
 60      * 
 61      * @internal
 62      * @since       3.0.4
 63      * @since       3.2.0   Changed the scope to public from protected as the head tag object will access it.
 64      */
 65     public function _isInThePage() {
 66         
 67         // If it's not in one of the post type's pages
 68         if ( ! $this->oProp->bIsAdmin ) {
 69             return false;
 70         }
 71         if ( ! in_array( $this->oProp->sPageNow, array( 'edit.php', 'edit-tags.php', 'post.php', 'post-new.php' ) ) ) {
 72             return false;
 73         }
 74                 
 75         return ( $this->oUtil->getCurrentPostType() == $this->oProp->sPostType );

154     public function _replyToRegisterTaxonomies() {

156         foreach( $this->oProp->aTaxonomies as $_sTaxonomySlug => $_aArgs ) {
157             $_aObjectTypes = is_array( $this->oProp->aTaxonomyObjectTypes[ $_sTaxonomySlug ] ) ? $this->oProp->aTaxonomyObjectTypes[ $_sTaxonomySlug ] : array();
158             $_aObjectTypes[] = $this->oProp->sPostType;
159             register_taxonomy(
160                 $_sTaxonomySlug,
161                 array_unique( $_aObjectTypes ), // object types
