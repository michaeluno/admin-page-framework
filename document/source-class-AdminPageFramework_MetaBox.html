168     /**
169     * Adds the given field array items into the field array property.
170     * 
171     * Identical to the addSettingFields() method except that this method does not accept enumerated parameters. 
172     * 
173     * @since            2.1.2
174     * @since            3.0.0           The scope changed to public to indicate the users will use.
175     * @return           void

177     public function addSettingField( array $aField ) {
178         
179         $aField = array( '_field_type' => 'post_meta_box' ) + $aField + AdminPageFramework_Property_MetaBox::$_aStructure_Field;    // avoid undefined index warnings.
180         
181         // Sanitize the IDs since they are used as a callback method name.
182         $aField['field_id'] = $this->oUtil->sanitizeSlug( $aField['field_id'] );
183         
184         // Check the mandatory keys are set.
185         if ( ! isset( $aField['field_id'], $aField['type'] ) ) return;  // these keys are necessary.
186                         
187         // If a custom condition is set and it's not true, skip.
188         if ( ! $aField['if'] ) return;
189                             
190         // Load head tag elements for fields.
191         if ( $this->_isPostDefinitionPage() ) 
192             AdminPageFramework_FieldTypeRegistration::_setFieldHeadTagElements( $aField, $this->oProp, $this->oHeadTag );   // Set relevant scripts and styles for the input field.
193         
194         // For the contextual help pane,
195         if ( $this->_isPostDefinitionPage() && $aField['help'] )
196             $this->oHelpPane->_addHelpTextForFormFields( $aField['title'], $aField['help'], $aField['help_aside'] );
197                 
198         $this->oProp->aFields[ $aField['field_id'] ] = $aField;
199     
200     }

202         /**
203          * Checks if the current page is post editing page or not.
204          * @since           3.0.0
205          */
206         protected function _isPostDefinitionPage() {
207             
208             if ( ! in_array( $GLOBALS['pagenow'], array( 'post.php', 'post-new.php', ) ) ) return false;
209             
210             if ( isset( $_GET['post_type'] ) && in_array( $_GET['post_type'], $this->oProp->aPostTypes ) )
211                 return true;
212             
213             $this->_sCurrentPostType = isset( $this->_sCurrentPostType )
214                 ? $this->_sCurrentPostType
215                 : ( isset( $_GET['post'] )
216                     ? get_post_type( $_GET['post'] )
217                     : ''
218                 );
219             
220             if ( isset( $_GET['post'], $_GET['action'] ) && in_array( $this->_sCurrentPostType, $this->oProp->aPostTypes ) )        // edit post page
221                 return true;                        
222             
223             return false;
224             
225         }
